add_library(${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL} SHARED
	src/gl_api.cpp
	src/gl_export.h
)

target_compile_definitions(${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL}
	PUBLIC
	REF_OPENGL_EXPERIMENTAL_ENABLED

	PRIVATE
	REF_DLL
)

set_common_library_compiler_settings(${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL})

target_include_directories(${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL} PRIVATE
	src
)

target_link_libraries(${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL} PRIVATE
	${TARGETNAME_LIB_ENGINEPUBLICAPI}
	${TARGETNAME_LIB_ENGINEINTERNALAPI}

	# Linux
	$<$<STREQUAL:"${TARGET_PLATFORM}","LINUX">:m>
)

install(TARGETS ${TARGETNAME_LIB_REF_OPENGL_EXPERIMENTAL}
	ARCHIVE DESTINATION ${INSTALL_ROOT}
	LIBRARY DESTINATION ${INSTALL_ROOT}
	RUNTIME DESTINATION ${INSTALL_ROOT}
)
